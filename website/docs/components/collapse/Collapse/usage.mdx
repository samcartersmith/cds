### Basic Collapse
```jsx live
function BasicCollapse() {
  const [expanded, { toggle }] = useToggler();

  return (
    <ExampleWithThemeToggles>
      <Button onPress={toggle}>Click me!</Button>
      <Collapse expanded={expanded}>
        <TextInput
          compact
          type="number"
          step="0.01"
          label="Amount"
          placeholder="8293323.23"
          suffix="USD"
        />
      </Collapse>
    </ExampleWithThemeToggles>
  )
}
```
