### Basic example

```jsx live
function Example() {
  const cardProps = {
    title: 'Send a crypto gift',
    description: 'Share the gift of crypto this holiday season',
    spotSquare: 'coinbaseCardSparkle',
    action: 'Start gifting',
  };
  return (
    <ResponsiveArtboard>
      <CardGroup>
        <FeatureEntryCard {...cardProps} />
        <FeatureEntryCard {...cardProps} />
      </CardGroup>
    </ResponsiveArtboard>
  );
}
```

### With divider

In Frontier a CardGroup will include a Divider component between each item.
You can override this by passing in `divider={null}`

```jsx live
function Example() {
  const cardProps = {
    title: 'Send a crypto gift',
    description: 'Share the gift of crypto this holiday season',
    spotSquare: 'coinbaseCardSparkle',
    action: 'Start gifting',
  };
  return (
    <ResponsiveArtboard>
      <CardGroup divider={null}>
        <FeatureEntryCard {...cardProps} />
        <FeatureEntryCard {...cardProps} />
      </CardGroup>
    </ResponsiveArtboard>
  );
}
```

### With custom divider

```jsx live
function Example() {
  const cardProps = {
    title: 'Send a crypto gift',
    description: 'Share the gift of crypto this holiday season',
    spotSquare: 'coinbaseCardSparkle',
    action: 'Start gifting',
  };

  const CustomDivider = useCallback(
    () => <Box height={4} background="backgroundAlternate" width="100%" />,
    [],
  );

  return (
    <ResponsiveArtboard>
      <CardGroup divider={CustomDivider}>
        <FeatureEntryCard {...cardProps} />
        <FeatureEntryCard {...cardProps} />
      </CardGroup>
    </ResponsiveArtboard>
  );
}
```

### With custom renderItem callback

If you need to add something custom to one of the items in a CardGroup you can use the renderItem callback to conditionally handle this.

```jsx live
function Example() {
  const cardProps = {
    title: 'Send a crypto gift',
    description: 'Share the gift of crypto this holiday season',
    spotSquare: 'coinbaseCardSparkle',
    action: 'Start gifting',
  };

  const customRenderItem = useCallback(({ item, Wrapper, isFirst, isLast }) => {
    return (
      <Wrapper>
        {isFirst && <Divider color="line" />}
        {item}
        {isLast && <Divider color="line" />}
      </Wrapper>
    );
  }, []);

  return (
    <ResponsiveArtboard>
      <CardGroup renderItem={customRenderItem}>
        <FeatureEntryCard {...cardProps} />
        <FeatureEntryCard {...cardProps} />
      </CardGroup>
    </ResponsiveArtboard>
  );
}
```
