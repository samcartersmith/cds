### Password input

Password Input - Use Icon Buttons at the end for actions like showing a password or clearing text from an input

```jsx live
function PasswordInput() {
  const [isVisible, { toggle: toggleIsVisible }] = useToggler();
  const [type, setType] = useState('text');

  useEffect(() => {
    setType(isVisible ? 'text' : 'password');
  }, [isVisible]);

  return (
    <ExampleWithThemeToggles>
      <TextInput
        label="Password"
        type={type}
        end={
          <IconButton
            transparent
            name={isVisible ? 'visibleActive' : 'visibleInactive'}
            variant="primary"
            onPress={toggleIsVisible}
          />
        }
      />
    </ExampleWithThemeToggles>
  );
}
```

### Link + Icon Button

If needed, you can add a Link + Icon Button like this example here. Use this sparingly and only at the End of an Input.

```jsx live
function CopyTextField() {
  const [copied, { toggleOff: toggleCopiedOff, toggle: toggleCopied }] = useToggler();
  const [variant, setVariant] = useState('foregroundMuted');
  const [helperText, setHelperText] = useState('');

  useEffect(() => {
    if (copied) {
      setVariant('positive');
      setHelperText('Your token has been copied!');
    } else {
      setVariant('foregroundMuted');
      setHelperText('');
    }
  }, [copied]);

  const handleOnChange = useCallback(() => {
    setVariant('foregroundMuted');
    toggleCopiedOff();
    setHelperText('');
  }, [toggleCopiedOff]);

  return (
    <ExampleWithThemeToggles>
      <TextInput
        end={
          <HStack>
            <Link onPress={toggleCopied} variant="caption" color={variant}>
              {copied ? 'copied' : 'copy'}
            </Link>
            <Icon spacingHorizontal={1} color="primary" name="visibleActive" size="s" />
          </HStack>
        }
        onChange={handleOnChange}
        variant={variant}
        helperText={helperText}
        label="API Access Token"
      />
    </ExampleWithThemeToggles>
  );
}
```

### Example of a Form

