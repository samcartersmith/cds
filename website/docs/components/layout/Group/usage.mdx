### With custom renderItem callback

```jsx live
function Example() {
  const CustomDivider = (
    <Box spacingBottom={gutter}>
      <Box height={4} background="primary" width="100%" />
    </Box>
  );

  const customRenderItem = useCallback(({ item, Wrapper, isFirst, isLast }) => {
    return (
      <Wrapper borderColor="lineHeavy" borderedTop={isFirst} borderedBottom={isLast}>
        {item}
      </Wrapper>
    );
  }, []);
  return (
    <Group gap={gutter} divider={CustomDivider} renderItem={customRenderItem}>
      <ExampleBox>Box 1</ExampleBox>
      <ExampleBox>Box 2</ExampleBox>
      <ExampleBox>Box 3</ExampleBox>
    </Group>
  );
}
```
