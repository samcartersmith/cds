A specialized box that that is optionally Pressable. Styled with rounded corners and a border and defaults to an elevation of 1.

## Clickable Cards

```jsx live
function Example() {
  const onPressConsole = () => console.log('pressed');
  const sharedProps = { spacing: 2 };
  const sharedPressProps = { onPress: onPressConsole, ...sharedProps };

  return (
    <HStack gap={2}>
      <Card {...sharedPressProps} elevation={1}>
        <TextLabel1 as="p" spacingBottom={1}>
          Elevation 1
        </TextLabel1>
        <TextBody as="p" spacingBottom={3}>
          {loremIpsum}
        </TextBody>
      </Card>

      <Card {...sharedPressProps} elevation={2}>
        <TextLabel1 as="p" spacingBottom={1}>
          Elevation 2
        </TextLabel1>
        <TextBody as="p" spacingBottom={3}>
          {loremIpsum}
        </TextBody>
      </Card>
    </HStack>
  );
}
```

## Pinned - top

```jsx live
function Example() {
  const sharedProps = { spacing: 2 };
  const pinnedSharedProps = { ...sharedProps, elevation: 2 };
  const pinnedSharedWrapperProps = {
    position: 'relative',
    background: 'backgroundAlternate',
    width: '100%',
    height: 100,
  };

  return (
    <Box {...pinnedSharedWrapperProps}>
      <Card {...pinnedSharedProps} pin="top">
        <TextLabel1 as="p">Top</TextLabel1>
      </Card>
    </Box>
  );
}
```
