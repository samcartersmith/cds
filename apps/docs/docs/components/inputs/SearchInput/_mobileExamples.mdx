### Basic usage

```tsx
function BasicSearchInput() {
  const [value, setValue] = useState('');
  return (
    <VStack gap={2}>
      <SearchInput
        value={value}
        onChangeText={setValue}
        onClear={() => setValue('')}
        onSearch={(searchTerm) => console.log('Searching for:', searchTerm)}
        placeholder="Search..."
      />
      <Text color="foregroundMuted" font="caption">
        Submit on keyboard or tap search icon
      </Text>
    </VStack>
  );
}
```

### With Back Navigation

```tsx
function SearchInputWithBack() {
  const [value, setValue] = useState('');
  return (
    <VStack gap={2}>
      <SearchInput
        value={value}
        onChangeText={setValue}
        onClear={() => setValue('')}
        onBack={() => console.log('Navigating back')}
        onSearch={(searchTerm) => console.log('Searching for:', searchTerm)}
        placeholder="Focus to see back arrow..."
      />
      <Text color="foregroundMuted" font="caption">
        Search icon changes to back arrow on focus
      </Text>
    </VStack>
  );
}
```

### Variants

When `bordered={false}`, SearchInput keeps focused border styling disabled by default. Set
`focusedBorderWidth` to opt into a focus border style.

```tsx
function SearchInputVariants() {
  const [value1, setValue1] = useState('');
  const [value2, setValue2] = useState('');
  const [value3, setValue3] = useState('');
  const [value4, setValue4] = useState('');

  return (
    <VStack gap={2}>
      <SearchInput
        value={value1}
        onChangeText={setValue1}
        onClear={() => setValue1('')}
        placeholder="Always show search icon..."
        disableBackArrow
      />
      <SearchInput
        value={value2}
        onChangeText={setValue2}
        onClear={() => setValue2('')}
        placeholder="Borderless search (default focus behavior)..."
        bordered={false}
      />
      <SearchInput
        value={value3}
        onChangeText={setValue3}
        onClear={() => setValue3('')}
        placeholder="Borderless search (with focus border)..."
        bordered={false}
        focusedBorderWidth={200}
      />
      <SearchInput
        value={value4}
        onChangeText={setValue4}
        onClear={() => setValue4('')}
        placeholder="No icons..."
        hideStartIcon
        hideEndIcon
      />
    </VStack>
  );
}
```
