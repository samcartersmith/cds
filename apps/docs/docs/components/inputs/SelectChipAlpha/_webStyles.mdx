import { useMemo, useState } from 'react';
import { assets } from '@coinbase/cds-common/internal/data/assets';
import { useMultiSelect } from '@coinbase/cds-common/select/useMultiSelect';
import { ComponentStylesTable } from '@site/src/components/page/ComponentStylesTable';
import { StylesExplorer } from '@site/src/components/page/StylesExplorer';
import { SelectChip } from '@coinbase/cds-web/alpha/select-chip';
import { RemoteImage } from '@coinbase/cds-web/media';
import { RemoteImageGroup } from '@coinbase/cds-web/media';

import webStylesData from ':docgen/web/alpha/select-chip/SelectChip/styles-data';

export const SelectChipBasicExample = ({ classNames }) => {
  const exampleOptions = [
    {
      label: 'Group A',
      options: [
        { value: null, label: 'Clear selection' },
        { value: '1', label: 'Option 1' },
        { value: '2', label: 'Option 2' },
        { value: '3', label: 'Option 3' },
      ],
    },
    {
      label: 'Group B',
      options: [
        { value: '4', label: 'Option 4' },
        { value: '5', label: 'Option 5' },
      ],
    },
  ];
  const [value, setValue] = useState(null);
  return (
    <SelectChip
      accessibilityLabel="Select a value"
      classNames={classNames}
      onChange={setValue}
      options={exampleOptions}
      placeholder="Choose an option"
      value={value}
    />
  );
};

export const SelectChipWithAssetsExample = ({ classNames }) => {
  const assetImageMap = {
    btc: assets.btc.imageUrl,
    eth: assets.eth.imageUrl,
    dai: assets.dai.imageUrl,
    ltc: assets.ltc.imageUrl,
    xrp: assets.xrp.imageUrl,
  };

const exampleOptions = [
{ value: 'btc', label: assets.btc.name },
{ value: 'eth', label: assets.eth.name },
{ value: 'dai', label: assets.dai.name },
{ value: 'ltc', label: assets.ltc.name },
{ value: 'xrp', label: assets.xrp.name },
];
const { value, onChange } = useMultiSelect({
initialValue: ['eth', 'btc'],
});

const startNode = useMemo(() => {
if (value.length === 0) return null;
return (

<RemoteImageGroup shape="circle" size={24}>
  {value.map((assetValue) => {
    const imageUrl = assetImageMap[assetValue];
    if (!imageUrl) return null;
    return <RemoteImage key={assetValue} source={imageUrl} />;
  })}
</RemoteImageGroup>
); }, [value]);

return (

<SelectChip
  controlAccessibilityLabel="Select multiple assets"
  classNames={classNames}
  maxWidth={400}
  onChange={onChange}
  options={exampleOptions}
  placeholder="Choose assets"
  startNode={startNode}
  type="multi"
  value={value}
/>
); };
