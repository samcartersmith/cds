These are the new navigation components. Read about the release process and motivation at [go/cds-nav-doc](http://go/cds-nav-doc).

## NavigationBar

Until the Navigation system and all it's parts are locked in, NavigationBar
is simply providing a bit of structure.

```jsx

  pageTitle={pageTitle}
  avatarUrl={user.photoUrl}
  onBackPress={canGoBack && handleBack}
  onMenuOpen={handleMenuOpen}
  onAppSwitch={handleAppSwitch}
/>
```

function fullExample() {
  const [activeIndex, setActiveIndex] = useState(0);

  const { name, avatar } = mocks.users[1];

  return (
    <ExampleWithThemeToggles>
      <HStack>
        <Sidebar
          logo={
            <Link to="/">
              <LogoMark />

          }
        >
          {items.map((props, index) => (
            <SidebarItem
              key={`example--sidebar-item--${props.title}`}
              active={index === activeIndex}
              onPress={() => setActiveIndex(index)}
              {...props}
            />
          ))}
        </Sidebar>
        <VStack>
          <NavigationBar
            end={
              <HStack gap={1} alignItems="center">
                <IconButton name="bell" onPress={() => handlePress('Notifications')} />
                <Pressable
                  backgroundColor="transparent"
                  onPress={() => handlePress('Avatar group')}
                >
                  <HStack gap={1} alignItems="center">
                    <Avatar alt="Kat" src={avatar} size="xl" />
                    <TextHeadline as="h2">{name}</TextHeadline>
                  </HStack>
                </Pressable>
              </HStack>
            }
          >
            <NavigationTitle>Personal Portfolio</NavigationTitle>
          </NavigationBar>
          <VStack spacing={2}>
            <TextDisplay2 as="h2" spacingBottom={1}>
              {items[activeIndex].title}
            </TextDisplay2>
            {loremIpsum}
            {loremIpsum}
            {loremIpsum}
            {loremIpsum}
            {loremIpsum}
            {loremIpsum}
            {loremIpsum}
          </VStack>
        </VStack>
      </HStack>
    </ExampleWithThemeToggles>
  );
}
