# Select Prototype

## Filter Example

```jsx live
function DefaultSelect() {
  const [year, setYear] = useState();
  const [asset, setAsset] = useState();
  const [showResult, toggleShowResult] = useToggler(false);

  const years = ['2015', '2016', '2017', '2018', '2019', '2020', '2021'];
  const currencies = [
    'Bitcoin',
    'Litecoin',
    'Ethereum',
    'Algorand',
    'Cardano',
    'Doge',
    'AMP',
    'Arpa',
  ];

  const handleButtonPress = useCallback(() => {
    if (year && asset) {
      toggleShowResult.toggleOn();
    }
  }, [year, asset, toggleShowResult]);
  return (
    <ExampleWithThemeToggles>
      <VStack gap={2} minHeight={500}>
        <HStack gap={1}>
          <Select width="100%" value={year} onChange={setYear} placeholder="Year">
            {years.map((option) => (
              <SelectOption
                value={option}
                key={option}
                title={option}
                description="BTC"
                selected={option === year}
              />
            ))}
          </Select>
          <Select width="100%" value={asset} onChange={setAsset} placeholder="Assets">
            {currencies.map((option) => (
              <SelectOption
                value={option}
                key={option}
                title={option}
                description="BTC"
                selected={option === asset}
              />
            ))}
          </Select>
        </HStack>
        <Button onPress={handleButtonPress}>Submit</Button>
        {showResult && (
          <HStack background="primaryWash" height={200} justifyContent="center" alignItems="center">
            <TextBody align="center" as="p">
              You have chosen {asset} from {year}
            </TextBody>
          </HStack>
        )}
      </VStack>
    </ExampleWithThemeToggles>
  );
}
```

function FormExample() {
  const accountTypeOptions = ['Savings Account', 'Checking Account'];

  const [accountType, setAccountType] = useState(accountTypeOptions[0]);
  const [visible, togglePopoverMenuVisibility] = useToggler(false);

  const handleAccountTypeChange = useCallback(
    (newValue) => {
      setAccountType(newValue);
    },
    [setAccountType],
  );

  const handleSubmit = useCallback((event) => {
    event.preventDefault();
    alert('Submitted');
  }, []);

  return (
    <form onSubmit={handleSubmit} action={undefined}>
      <VStack gap={2} minHeight={700} background="background">
        <TextInput label="Account number" />
        <TextInput label="Re-enter account number" />
        <Select value={accountType} onChange={handleAccountTypeChange}>
          {accountTypeOptions.map((option) => (
            <SelectOption
              value={option}
              key={option}
              description={option}
              selected={accountType === option}
            />
          ))}
        </Select>
        <Button type="submit">Submit</Button>
      </VStack>

  );
}
