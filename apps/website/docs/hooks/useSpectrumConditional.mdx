> [@cbhq/cds-common/hooks/useSpectrumConditional](https://github.cbhq.net/frontend/cds/blob/master/packages/common/hooks/useSpectrumConditional.ts)

Conditionally return a value based on the active spectrum.

```jsx live
function Example() {
  const size = 100;
  const paletteOverride = useSpectrumConditional({
    light: { primary: 'blue60' },
    dark: { primary: 'red60' },
  });
  return (
    <ExampleWithThemeToggles>
      <ThemeProvider palette={paletteOverride}>
        <Box background="primary" width={size} height={size} />
      </ThemeProvider>
    </ExampleWithThemeToggles>
  );
}
```
