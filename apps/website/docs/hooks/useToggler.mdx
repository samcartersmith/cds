> [@cbhq/cds-common/hooks/useToggler](https://github.cbhq.net/frontend/cds/blob/master/packages/common/hooks/useToggler.ts)

Hook for managing boolean state. Provides a set of functions for updating the state. Useful for managing visibility of components. Returns an array with the current state and an object with the following functions:

- toggleOn
- toggleOff
- toggle

```jsx live
function Example() {
  const [isOpen, { toggleOn, toggleOff, toggle }] = useToggler();

  return (
    <VStack gap={3}>
      <HStack gap={2}>
        <Button onClick={toggleOn}>Open</Button>
        <Button onClick={toggleOff}>Close</Button>
        <Button onClick={toggle}>Toggle Visibility</Button>
      </HStack>
      {isOpen && (
        <HStack spacing={3} borderRadius="rounded" background="primaryWash">
          <TextBody as="p">Hello world!</TextBody>
        </HStack>
      )}
    </VStack>
  );
}
```
