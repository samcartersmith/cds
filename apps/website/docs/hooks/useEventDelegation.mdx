> [@cbhq/cds-(web|mobile)/hooks/useEventDelegation](https://github.cbhq.net/frontend/cds/blob/master/packages/common/system/useEventDelegation.ts)

Returns a callback to capture a specific component event if specified in [EventDelegationProvider config](/components/event-delegation-provider).

In order to use this hook, EventDelegationProvider must wrap your root application component.

## Examples

```jsx
function Example(props) {
  const customEventConfig = {
    actions: ['onPress'],
    component_name: 'trade',
    custom_data: {
      side: 'buy',
    },
  };

  const onPressEvent = useEventDelegation('Button', 'onPress', customEventConfig);

  const onPress = useCallback(
    (event: React.MouseEvent) => {
      onPressEvent();
      props.onPress?.(event);
    },
    [onPressEvent, props.onPress],
  );

  return <Pressable onPress={onPress}>Press me</Pressable>;
}
```
