> [@cbhq/cds-(web|mobile)/hooks/useEventHandler](https://github.cbhq.net/frontend/cds/blob/master/packages/common/system/useEventHandler.ts)

Returns a callback to capture a specific component event if specified in [EventHandlerProvider config](/components/event-handler-provider).

In order to use this hook, EventHandlerProvider must wrap your root application component.

## Examples

```jsx
function Example(props) {
  const customEventConfig = {
    actions: ['onPress'],
    component_name: 'trade',
    custom_data: {
      side: 'buy',
    },
  };

  const onPressEvent = useEventHandler('Button', 'onPress', customEventConfig);

  const onPress = useCallback(
    (event: React.MouseEvent) => {
      onPressEvent();
      props.onPress?.(event);
    },
    [onPressEvent, props.onPress],
  );

  return <Pressable onPress={onPress}>Press me</Pressable>;
}
```
