This recipe offers guidance on how you could compose a system of Navigation components in an application. This recipe is for demonstration purposes only and uses hard coded data. Please note: the App and User Switcher components in this recipe are recommendations on how to compose them. Identity will own these components long term.

```jsx live
function NavigationRecipe() {
  const [isCollapsed, toggleCollapsed] = useToggler(false);
  const [activeIndex, setActiveIndex] = useState(0);
  const [moreMenuValue, setMoreMenuValue] = useState(undefined);

  // Sidebar config
  const SIDEBAR_ITEMS_TO_SHOW = 4;
  const sidebarItems = [
    { title: 'Assets', icon: 'chartPie' },
    { title: 'Trade', icon: 'trading' },
    { title: 'Pay', icon: 'pay' },
    { title: 'For you', icon: 'newsfeed' },
    { title: 'Earn', icon: 'giftBox' },
    { title: 'Borrow', icon: 'cash' },
    { title: 'DeFi', icon: 'defi' },
  ];
  const navItems = sidebarItems.slice(0, SIDEBAR_ITEMS_TO_SHOW);
  const moreMenuOptions = sidebarItems.slice(SIDEBAR_ITEMS_TO_SHOW);

  // Tabs config
  const tabs = [
    {
      id: 'all',
      label: 'All',
    },
    {
      id: 'watchlist',
      label: 'Watchlist',
    },
    {
      id: 'tradable',
      label: 'Tradable',
    },
    {
      id: 'gainers',
      label: 'Gainers',
      count: 2,
    },
    {
      id: 'losers',
      label: 'Losers',
    },
    {
      id: 'trending',
      label: 'Trending',
    },
    {
      id: 'other',
      label: 'Other',
    },
  ];

  const [value, setValue] = useState(tabs[1].id);
  const showBackButton = useMemo(() => value !== tabs[0].id, [value]);
  const handleBackPress = useCallback(() => {
    setValue(tabs[0].id);
  }, []);

  // NOTE: This is for example purposes only. Identity will create a UserSwitcher component.
  const UserSwitcherSubject = ({ title, description }) => {
    return (
      <Pressable
        as="button"
        backgroundColor="transparent"
        block
        borderRadius="rounded"
        noScaleOnPress
      >
        <HStack gap={1} alignItems="center">
          <Avatar size="xl" alt="Test" />
          <VStack maxWidth={172} minWidth={0}>
            {description ? (
              <TextLabel1 overflow="truncate" as="p">
                {title}
              </TextLabel1>
            ) : (
              <TextHeadline overflow="truncate" as="h3">
                {title}
              </TextHeadline>
            )}
            {description ? (
              <TextLabel2 overflow="truncate" as="p">
                {description}
              </TextLabel2>
            ) : null}
          </VStack>
        </HStack>
      </Pressable>
    );
  };

  const handleMoreMenuChange = (newValue) => {
    const moreIndex =
      moreMenuOptions.findIndex((option) => option.title === newValue) + navItems.length;
    setActiveIndex(moreIndex);
    setMoreMenuValue(newValue);
  };

  const handleItemPress = (index) => {
    setActiveIndex(index);
    setMoreMenuValue(undefined);
  };

  const switcherPositionConfig = {
    placement: 'bottom',
    gap: 1,
  };

  return (
    <PortalProvider>
      <FeatureFlagProvider frontierColor frontierButton>
        <HStack>
          <Sidebar collapsed={isCollapsed} logo={<LogoMark />}>
            {navItems.map((props, index) => (
              <SidebarItem
                key={`sidebar-item--${props.title}`}
                active={index === activeIndex}
                onPress={() => handleItemPress(index)}
                tooltipContent={props.title}
                {...props}
              />
            ))}
            <SidebarMoreMenu
              onChange={handleMoreMenuChange}
              value={moreMenuValue}
              active={activeIndex >= navItems.length}
              tooltipContent="More"
            >
              {moreMenuOptions.map((item) => (
                <SelectOption
                  key={`sidebar-more-menu-item--${item.title}`}
                  value={item.title}
                  description={item.title}
                  media={<NavigationIcon name={item.icon} />}
                />
              ))}
            </SidebarMoreMenu>
          </Sidebar>
          <VStack overflow="clip">
            <NavigationBar
              start={showBackButton && <IconButton name="backArrow" onPress={handleBackPress} />}
              end={
                <HStack alignItems="center" justifyContent="flex-end" gap={1}>
                  {/* NOTE: this is for example purposes only. Identity will create the AppSwitcher components and own them. */}
                  <Dropdown
                    width={359}
                    maxHeight={600}
                    showOverlay
                    enableMobileModal
                    contentPosition={switcherPositionConfig}
                    content={<AppSwitcherContent />}
                  >
                    <NavigationIconButton
                      accessibilityLabel="App Switcher Menu"
                      name="appSwitcher"
                    />
                  </Dropdown>
                  {/* NOTE: this is for example purposes only. Identity will create the UserSwitcher components and own them. */}
                  <Dropdown
                    content={<UserSwitcherContent />}
                    width={400}
                    maxHeight={600}
                    showOverlay
                    contentPosition={switcherPositionConfig}
                    showMobileModal
                  >
                    <UserSwitcherSubject title="Brian Armstrong" />
                  </Dropdown>
                </HStack>
              }
              bottom={<TabNavigation tabs={tabs} value={value} onChange={setValue} />}
            >
              <NavigationTitle>{sidebarItems[activeIndex].title}</NavigationTitle>
            </NavigationBar>
            <HStack
              justifyContent="center"
              alignItems="flex-start"
              background="backgroundAlternate"
              flexGrow={1}
              spacing={4}
            >
              <Button compact variant="primary" onPress={toggleCollapsed.toggle}>
                {isCollapsed ? 'Expand' : 'Collapse'} Sidebar
              </Button>
            </HStack>
          </VStack>
        </HStack>
      </FeatureFlagProvider>
    </PortalProvider>
  );
}
```
