```tsx
import { DevicePreferencesProvider } from '@cbhq/cds-web-esm/system/DevicePreferencesProvider';
import { FeatureFlagProvider } from '@cbhq/cds-web-esm/system/FeatureFlagProvider';
import { ThemeProvider } from '@cbhq/cds-web-esm/system/ThemeProvider';
import { PortalProvider } from '@cbhq/cds-web-esm/overlays/PortalProvider';

function App() {
  return (
    <FeatureFlagProvider>
      <DevicePreferencesProvider>
        // ThemeProvider must be a child of FeatureFlagProvider & DevicePreferencesProvider.
        <ThemeProvider>
          // PortalProvider must be a child of ThemeProvider
          <PortalProvider>
            <ErrorBoundary>
              <AppRoutes />
            </ErrorBoundary>
          </PortalProvider>
        </ThemeProvider>
      </DevicePreferencesProvider>
    </FeatureFlagProvider>
  );
}
```
