### renderEnd

We provide a `renderEnd` prop that will render a node fixed to the bottom of the sidebar. Please pay close attention to collapsed and default states. This is a render prop - it passes the sidebars `collapsed` state so you can properly setup your UI. Please ensure you're either providing a UI that works well in the collapsed state, or hiding your UI when collapsed.

:::note
Always hoist or memoize your responsive style configurations.
:::

```jsx live
function SidebarItemWithPress() {
  const [activeIndex, setActiveIndex] = useState(0);
  const items = mocks.product.navigationItems;

  const handlePress = (index) => {
    setActiveIndex(index);
  };

  // Setup your end node ouside the component
  // ----------------------------------------
  const renderCB1 = (isCollapsed) => {
    const spacing = isCollapsed ? 1 : 2;

    return (
      <Pressable
        onPress={() => handlePress('Notifications')}
        as="button"
        borderRadius="roundedLarge"
        backgroundColor="primaryWash"
      >
        <HStack gap={1} alignItems="center" justifyContent="center" spacing={spacing}>
          <Pictogram name="coinbaseOneLogoPictogram" dimension="48x48" scaleMultiplier={0.8} />
          {!isCollapsed && (
            <VStack>
              <TextHeadline as="span">Coinbase One</TextHeadline>
              <TextLegal as="p" color="foregroundMuted">
                Zero trading fees
              </TextLegal>
            </VStack>
          )}
        </HStack>
      </Pressable>
    );
  };

  return (
    <ExampleWithThemeToggles>
      <HStack
        justifyContent="center"
        alignItems="flex-start"
        dangerouslySetBackground={palette.backgroundAlternate}
      >
        <Sidebar autoCollapse logo={<LogoMark />} renderEnd={renderCB1}>
          {items.map((props, index) => (
            <SidebarItem
              key={`sidebar-item--${props.title}`}
              active={index === activeIndex}
              onPress={() => handlePress(index)}
              {...props}
            />
          ))}
        </Sidebar>
        <HStack spacing={2} gap={1} justifyContent="space-between" alignItems="center" flexGrow={1}>
          <TextHeadline as="h2">Active Page: {items[activeIndex].title}</TextHeadline>
        </HStack>
      </HStack>
    </ExampleWithThemeToggles>
  );
}
```
