import { ApiDetails } from ':cds-website/components/ApiDetails';
import { Badge } from ':cds-website/components/Badge';

## Props

### alt

This is the name associated with the entity in the Avatar. This is used in the image alt tag for accessibility.

<ApiDetails
  componentName="Avatar"
  propertyName="alt"
  required={true}
  defaultValue=""
  mobileOptions={[]}
  webOptions={[]}
/>

### borderColor

Adds a border to the Avatar

<ApiDetails
  componentName="Avatar"
  propertyName="borderColor"
  required={false}
  defaultValue=""
  mobileOptions={[
    'line',
    'lineHeavy',
    'negative',
    'positive',
    'primary',
    'primaryWash',
    'secondary',
    'transparent',
  ]}
  webOptions={[
    'line',
    'lineHeavy',
    'negative',
    'positive',
    'primary',
    'primaryWash',
    'secondary',
    'transparent',
  ]}
/>

### dangerouslySetClassName <Badge variant="web" /> <Badge variant="danger" order={2} />

Adds a className to the Avatar. If you pass in a className make sure your styles override the Avatar using the avatar class .cds-avatar like this: .my-class.cds-avatar

<ApiDetails
  componentName="Avatar"
  propertyName="dangerouslySetClassName"
  required={false}
  defaultValue=""
  mobileOptions={[]}
  webOptions={['string']}
/>

### dangerouslySetSize <Badge variant="danger" order={1} />

This is an escape hatch when using the Avatar in a fixed size container where you cannot control the dimensions.
Creates a custom Avatar size. The size prop should be used in most circumstances.

<ApiDetails
  componentName="Avatar"
  propertyName="dangerouslySetSize"
  required={false}
  defaultValue=""
  mobileOptions={['number']}
  webOptions={['number']}
/>

### selected <Badge variant="web" />

Adds treatment that indicates that the Avatar is currently selected

<ApiDetails
  componentName="Avatar"
  propertyName="selected"
  required={false}
  defaultValue=""
  mobileOptions={[]}
  webOptions={['false', 'true']}
/>

### shape

Shape of Avatar

<ApiDetails
  componentName="Avatar"
  propertyName="shape"
  required={false}
  defaultValue=""
  mobileOptions={['circle', 'square']}
  webOptions={['circle', 'square']}
/>

### size

Size for a given avatar.

<ApiDetails
  componentName="Avatar"
  propertyName="size"
  required={false}
  defaultValue=""
  mobileOptions={['l', 'm', 'xl', 'xxl', 'xxxl']}
  webOptions={['l', 'm', 'xl', 'xxl', 'xxxl']}
/>

### src

Absolute url to the image that should be shown in the Avatar. If no src is provided then a generic fallback image is used.

<ApiDetails
  componentName="Avatar"
  propertyName="src"
  required={false}
  defaultValue=""
  mobileOptions={['string']}
  webOptions={['string']}
/>

### testID

Used to locate this element in unit and end-to-end tests. Under the hood, testID translates to data-testid on Web. On Mobile, testID stays the same - testID

<ApiDetails
  componentName="Avatar"
  propertyName="testID"
  required={false}
  defaultValue=""
  mobileOptions={['string']}
  webOptions={['string']}
/>
