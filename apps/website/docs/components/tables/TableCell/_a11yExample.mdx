## Tables with two headers

The following table of asset prices has header information in both the top row and the first column. All header cells are marked up as `<th />` cells with scope attributes added. Read more about working with multiple headers on [w3.org](https://www.w3.org/WAI/tutorials/tables/two-headers/)

```jsx live
<Table maxHeight={300} bordered variant="ruled">
  <TableHeader sticky>
    <TableRow backgroundColor="backgroundAlternate">
      <TableCell title="Currency" />
      <TableCell title="Balance" />
      <TableCell title="Status" alignItems="flex-end" />
    </TableRow>
  </TableHeader>
  <TableBody>
    <TableRow>
      <TableCell title="BTC" as="th" />
      <TableCell title="$100" />
      <TableCell title="Pending" alignItems="flex-end" />
    </TableRow>
    <TableRow>
      <TableCell title="ETH" as="th" />
      <TableCell title="$200" />
      <TableCell title="Complete" alignItems="flex-end" />
    </TableRow>
    <TableRow>
      <TableCell title="APE" as="th" />
      <TableCell title="$300" />
      <TableCell title="Complete" alignItems="flex-end" />
    </TableRow>
    <TableRow>
      <TableCell title="SOL" as="th" />
      <TableCell title="$300" />
      <TableCell title="Complete" alignItems="flex-end" />
    </TableRow>
    <TableRow>
      <TableCell title="CVX" as="th" />
      <TableCell title="$300" />
      <TableCell title="Complete" alignItems="flex-end" />
    </TableRow>
    <TableRow>
      <TableCell title="AVX" as="th" />
      <TableCell title="$300" />
      <TableCell title="Complete" alignItems="flex-end" />
    </TableRow>
  </TableBody>
</Table>
```
