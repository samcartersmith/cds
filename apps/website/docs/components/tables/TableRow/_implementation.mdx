import { ApiDetails } from ':cds-website/components/ApiDetails';
import { Badge } from ':cds-website/components/Badge';

## Props

### backgroundColor <Badge variant="web" />

Set the background color for this entire row to some CDS Palette background color name

<ApiDetails
  componentName="TableRow"
  propertyName="backgroundColor"
  required={false}
  defaultValue="undefined"
  mobileOptions={[]}
  webOptions={[
    'background',
    'backgroundAlternate',
    'backgroundOverlay',
    'negative',
    'positive',
    'primary',
    'primaryWash',
    'secondary',
    'transparent',
  ]}
/>
