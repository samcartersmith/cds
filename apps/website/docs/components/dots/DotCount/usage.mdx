### Basic DotCount

```jsx live
function DotCountBasic() {
  const counts = [1, 100, 30, 2, 0, 99];
  return (
    <ResponsiveExample>
      {counts.map((count) => (
        <Box spacing={1} key={`DotCount-${count}`}>
          <DotCount count={count} />
        </Box>
      ))}
    </ResponsiveExample>
  );
}
```

### Placements

```jsx live
function Placements() {
  const PIN_DIRECTIONS = ['top-end'];
  return (
    <ResponsiveExample>
      {PIN_DIRECTIONS.map((pinDirection) => (
        <VStack gap={4} spacing={1} key={`DotCount-placement-${pinDirection}`} spacingBottom={4}>
          <DotCount pin={pinDirection} count={30}>
            <Avatar size="xxxl" shape="square" alt="Sneezy" />
          </DotCount>
          <DotCount pin={pinDirection} count={30}>
            <Avatar size="xl" shape="square" alt="Sneezy" />
          </DotCount>
          <DotCount pin={pinDirection} count={1}>
            <Icon name="airdrop" shape="square" size="l" />
          </DotCount>
        </VStack>
      ))}
    </ResponsiveExample>
  );
}
```
