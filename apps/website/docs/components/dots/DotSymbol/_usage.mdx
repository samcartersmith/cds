### Placements

```jsx live
function Placements() {
  const PIN_DIRECTIONS = ['top-start', 'top-end', 'bottom-start', 'bottom-end'];
  return (
    <ResponsiveExample>
      {PIN_DIRECTIONS.map((pinDirection) => (
        <VStack gap={4} spacing={1} key={`dotsymbol-placement-${pinDirection}`} spacingBottom={4}>
          <DotSymbol size="m" pin={pinDirection} source={assets.eth.imageUrl}>
            <Avatar size="xxxl" shape="square" alt="Sneezy" />
          </DotSymbol>
          <DotSymbol size="m" pin={pinDirection} source={assets.eth.imageUrl}>
            <Avatar size="xl" shape="square" alt="Sneezy" />
          </DotSymbol>
          <DotSymbol size="m" pin={pinDirection} source={assets.eth.imageUrl}>
            <Icon name="airdrop" shape="square" size="l" />
          </DotSymbol>
        </VStack>
      ))}
    </ResponsiveExample>
  );
}
```

### Sizes

```jsx live
function Sizes() {
  const SIZES = ['xs', 's', 'm', 'l'];
  return (
    <ResponsiveExample>
      {SIZES.map((size) => (
        <VStack gap={1} key={`dotsymbol-size-${size}`}>
          <DotSymbol source={assets.eth.imageUrl} size={size} />
        </VStack>
      ))}
    </ResponsiveExample>
  );
}
```

### Dotting relative to shape of child

You can use the overlap prop to place the dot relative to the corner of the wrapped element.

```jsx live
<ResponsiveExample>
  <HStack gap={2}>
    <DotSymbol source={assets.polygon.imageUrl} overlap="circular" pin="bottom-end" size="m">
      <Avatar src={assets.dai.imageUrl} size="xxxl" />
    </DotSymbol>
    <DotSymbol source={assets.polygon.imageUrl} overlap="circular" pin="bottom-end" size="l">
      <Avatar src={assets.dai.imageUrl} size="xxxl" />
    </DotSymbol>
  </HStack>
</ResponsiveExample>
```
