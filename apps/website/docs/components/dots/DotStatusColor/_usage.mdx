### Placements

```jsx live
function Variants() {
  const PIN_DIRECTIONS = ['top-start', 'top-end', 'bottom-start', 'bottom-end'];
  return (
    <ResponsiveExample>
      {PIN_DIRECTIONS.map((pinDirection) => (
        <VStack
          gap={4}
          spacing={1}
          key={`DotStatusColor-placement-${pinDirection}`}
          spacingBottom={4}
        >
          <DotStatusColor variant="positive" pin={pinDirection}>
            <Avatar size="xxxl" shape="square" alt="Sneezy" />
          </DotStatusColor>
          <DotStatusColor variant="positive" pin={pinDirection}>
            <Avatar size="xxl" shape="square" alt="Sneezy" />
          </DotStatusColor>
          <DotStatusColor variant="positive" pin={pinDirection}>
            <Icon name="airdrop" shape="square" size="l" />
          </DotStatusColor>
        </VStack>
      ))}
    </ResponsiveExample>
  );
}
```

### Sizes

```jsx live
function Sizes() {
  const SIZES = ['xs', 's', 'm', 'l'];
  return (
    <ResponsiveExample>
      {SIZES.map((size) => (
        <VStack gap={1} key={`DotStatusColor-size-${size}`}>
          <DotStatusColor variant="negative" size={size} />
        </VStack>
      ))}
    </ResponsiveExample>
  );
}
```

### Variants

```jsx live
function Placements() {
  const VARIANTS = ['positive', 'negative', 'primary', 'foregroundMuted'];
  return (
    <ExampleWithThemeToggles>
      {VARIANTS.map((variant) => (
        <VStack gap={1} key={`DotStatusColor-variant-${variant}`}>
          <DotStatusColor variant={variant} />
        </VStack>
      ))}
    </ExampleWithThemeToggles>
  );
}
```

### Dotting relative to shape of child

You can use the overlap prop to place the dot relative to the corner of the wrapped element.

```jsx live
<ResponsiveExample>
  <HStack gap={2}>
    <DotStatusColor variant="positive" pin="top-end">
      <RemoteImage size="xxxl" />
    </DotStatusColor>
    <DotStatusColor variant="negative" pin="top-end">
      <RemoteImage size="xxxl" />
    </DotStatusColor>
    <DotStatusColor variant="positive" overlap="circular" pin="top-end">
      <Avatar size="xxxl" />
    </DotStatusColor>
    <DotStatusColor variant="negative" overlap="circular" pin="top-end">
      <Avatar size="xxxl" />
    </DotStatusColor>
  </HStack>
</ResponsiveExample>
```
