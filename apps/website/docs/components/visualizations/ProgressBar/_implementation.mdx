## Props

### color

Custom progress color.

<ApiDetails
  componentName="ProgressBar"
  propertyName="color"
  required={false}
  defaultValue="primary"
  mobileOptions={['foreground', 'negative', 'positive', 'primary']}
  webOptions={['foreground', 'negative', 'positive', 'primary']}
/>

### disabled

Toggle used to show a disabled progress visualization

<ApiDetails
  componentName="ProgressBar"
  propertyName="disabled"
  required={false}
  defaultValue="false"
  mobileOptions={['false', 'true']}
  webOptions={['false', 'true']}
/>

<ApiDetails
  componentName="ProgressBar"

  componentName="ProgressBar"

  defaultValue=""

  webOptions={['string']}
/>
