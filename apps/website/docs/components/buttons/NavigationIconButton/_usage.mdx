`NavigationIconButton` is meant to be used inside of a `NavigationBar`. It is typically used to trigger the visibility of a `Dropdown`.

```jsx live
function NavigationMenu() {
  const [value, setValue] = useState('');
  const [visible, togglePopoverMenuVisibility] = useToggler(false);

  const navigationOptions = [
    {

    },
    {
      name: 'Wallet',
      value: 'wallet',
      description: 'The best self-hosted crypto wallet',
      mediaName: 'wallet',
    },
  ];

  const handleMenuValueChange = useCallback(
    (newValue) => {
      setValue(newValue);
    },
    [setValue],
  );

  return (
    <VStack minHeight={300} spacing={2} background="background">
      <NavigationBar
        end={
          <HStack gap={1}>
            <NavigationIconButton name="bell" variant="secondary" />
            <Dropdown
              width={350}
              content={
                <>

                </>
              }

              <NavigationIconButton name="appSwitcher" />
            </Dropdown>
          </HStack>
        }

}
```
