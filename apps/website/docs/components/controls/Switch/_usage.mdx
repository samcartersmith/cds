function SwitchDemo() {
  // doc site theme control
  const { colorMode, setColorMode } = useColorMode();
  // from @cbhq/cds-common/hooks/useToggler
  const [isChecked, { toggle }] = useToggler(colorMode === 'dark');

  const handleChange = useCallback(
    (event) => {
      setColorMode(event.target.checked ? 'light' : 'dark');
      toggle();
    },
    [setColorMode, toggle],
  );

  return (
    <ThemeProvider spectrum={isChecked ? 'dark' : 'light'}>
      <Switch onChange={handleChange} checked={isChecked}>
        Dark Mode
      </Switch>
    </ThemeProvider>
  );
}
