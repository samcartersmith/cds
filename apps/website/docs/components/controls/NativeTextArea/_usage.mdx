### Creating a CDS TextArea or Multiline TextInput

```jsx live noInline
const MAX_CHARS = 255;

function RenderNativeTextArea() {
  const [inputText, setInputText] = useState('');

  const onChange = useCallback((e) => {
    setInputText(e.target.value);
  }, []);

  const customNativeTextAreaCSS = useMemo(() => {
    return {
      resize: 'none',
    };
  }, []);

  const charsRemaining = useMemo(
    () => `${MAX_CHARS - inputText.length} characters remaining`,
    [inputText.length],
  );

  return (
    <Box spacing={1}>
      <TextInput
        inputNode={
          <NativeTextArea
            value={inputText}
            onChange={onChange}
            style={customNativeTextAreaCSS}
            rows={7}
            cols={5}
            placeholder="Type something here..."
            maxlength={MAX_CHARS}
          />
        }
        helperText={charsRemaining}
        label="TextArea with character counter"
      />
    </Box>
  );
}

render(<RenderNativeTextArea />);
```
