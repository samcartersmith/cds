### Creating a CDS TextArea or Multiline TextInput

```jsx live
function RenderNativeTextArea() {
  const [inputText, setInputText] = useState('Start writing something in the TextArea');

  const onChange = useCallback((e) => {
    setInputText(e.target.value);
  }, []);

  const customNativeTextAreaCSS = useMemo(() => {
    return {
      resize: 'none',
    };
  }, []);

  return (
    <Box spacing={1}>
      <TextInput
        inputNode={<NativeTextArea style={customNativeTextAreaCSS} rows={7} cols={5} />}
        onChange={onChange}
        helperText={inputText}
        label="Message Area"
      />
    </Box>
  );
}
```
