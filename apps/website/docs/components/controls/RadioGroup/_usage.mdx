function RadioGroupDemo() {
  // static values should live outside the component in product code.
  const options1 = {
    btc: 'Bitcoin',
    eth: 'Ethereum',
    dai: 'Dai',
  };

  const options2 = {
    'yellow-jacket': 'Yellow Jacket',
    bruin: 'Bruin',
    bronco: 'Bronco',
  };

  // example component using RadioGroup
  const [group1, setGroup1] = useState('btc');
  const [group2, setGroup2] = useState();
  const [group3, setGroup3] = useState();

  return (
    <>
      <RadioGroup
        aria-labelledby="choose-a-currency"
        id="currency-radio-group"
        label={
          <TextHeadline as="span" id="choose-a-currency">
            Choose a currency
          </TextHeadline>
        }
        name="radio-group1"
        selectedValue={group1}
        onChange={setGroup1}
        options={options1}
      />
      <TextHeadline as="span" id="choose-a-mascot">
        Choose a mascot
      </TextHeadline>
      <RadioGroup
        id="mascot-radio-group"
        accessibilityLabelledBy="choose-a-mascot"
        name="radio-group2"
        selectedValue={group2}
        onChange={setGroup2}
        options={options2}
      />
      <RadioGroup
        aria-labelledby="choose-a-currency2"
        id="horizontal-radio-group"
        label={
          <TextHeadline as="span" id="choose-a-currency2">
            Choose a currency
          </TextHeadline>
        }
        name="radio-group3"
        selectedValue={group3}
        onChange={setGroup3}
        options={options1}

      />

}
