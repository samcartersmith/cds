:::info
If you are wrapping a CDS `Modal` to create a custom Modal component, you will want to spread props `...props` or explicitly pass through `visible` and `onDidClose` to the underlying `Modal` component. Under the hood, we use `cloneElement` to pass `visible` and other handlers to the `Modal` component which are required for it to work.
:::

See example below:

```jsx live
function PortalModalExample() {
  const triggerRef = useRef(null); // only required for web

  function ExtractedModal({ visible, closeModal, onDidClose }) {
    const handleDidClose = useCallback(() => {
      if (onDidClose) {
        onDidClose();
      }

      if (triggerRef.current) {
        triggerRef.current.focus(); // only required for web
      }
    }, []);

    return (
      <Modal visible={visible} onRequestClose={closeModal} onDidClose={handleDidClose}>
        <ModalHeader title="Basic Modal" />
        <ModalBody>
          <TextBody as="p">{loremIpsum}</TextBody>
        </ModalBody>
        <ModalFooter primaryAction={<Button onPress={closeModal}>Save</Button>} />
      </Modal>
    );
  }

  function PortalModal() {
    const { openModal, closeModal } = useModal();

    const handlePress = useCallback(
      () => openModal(<ExtractedModal closeModal={closeModal} />),
      [openModal, closeModal],
    );

    return (
      <Box>
        <Button onPress={handlePress} ref={triggerRef}>
          Open Modal
        </Button>
      </Box>
    );
  }

  return (
    <PortalProvider>
      <ExampleWithThemeToggles>
        <PortalModal />
      </ExampleWithThemeToggles>
    </PortalProvider>
  );
}
```
