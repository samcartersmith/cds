Most likely you'd want to extract your Modal into a separate component.
You can do that by adding a `visible: boolean` and `onDidClose: () => void` prop to your extracted component and pass through those props to `Modal`.
See example below:

```jsx live
function PortalModalExample() {
  function ExtractedModal({ visible, closeModal, onDidClose }) {
    return (
      <Modal visible={visible} onRequestClose={closeModal} onDidClose={onDidClose}>
        <ModalHeader title="Basic Modal" />
        <ModalBody>
          <TextBody as="p">{loremIpsum}</TextBody>
        </ModalBody>
        <ModalFooter primaryAction={<Button onPress={closeModal}>Save</Button>} />
      </Modal>
    );
  }

  function PortalModal() {
    const { openModal, closeModal } = useModal();

    const handlePress = useCallback(
      () => openModal(<ExtractedModal closeModal={closeModal} />),
      [openModal, closeModal],
    );

    return (
      <Box>
        <Button onPress={handlePress}>Open Modal</Button>
      </Box>
    );
  }

  return (
    <PortalProvider>
      <ExampleWithThemeToggles>
        <PortalModal />
      </ExampleWithThemeToggles>
    </PortalProvider>
  );
}
```
