`AnimatedCaret` can be used to indicate that there is collapsed content.

```jsx live
function () {
    const [collapsed, { toggle }] = useToggler(true);

    return (
        <VStack background={collapsed ? 'background' : 'backgroundAlternate'} borderRadius="rounded">
            <PressableOpacity noScaleOnPress onPress={toggle}>
            <ListCell
                title="Title"
                description='Description'
                action={<AnimatedCaret rotate={collapsed ? 90 : 180} />}
            />
            </PressableOpacity>
            <Collapsible collapsed={collapsed} spacingHorizontal={3} spacingBottom={2}>
                <TextBody as="p">Peek a boo!</TextBody>
            </Collapsible>
        </VStack>
    )
}
```
