# Radio Group

Make sure that each radio group has their own focus.

export const options1 = {
  btc: 'Bitcoin',
  eth: 'Ethereum',
  dai: 'Dai',
};

export const options2 = {
  'yellow-jacket': 'Yellow Jacket',
  bruin: 'Bruin',
  bronco: 'Bronco',
};

<Canvas isColumn withToolbar>
  <Story name="Group">
    {() => {
      const [group1, setGroup1] = useState('btc');
      const [group2, setGroup2] = useState();
      return (
        <>
          <RadioGroup
            id="currency-radio-group"
            label={<TextHeadline as="h1">Choose a currency</TextHeadline>}
            name="radio-group1"
            selectedValue={group1}
            onChange={setGroup1}
            options={options1}
          />
          <TextHeadline as="h1" id="choose-a-mascot">
            Choose a mascot
          </TextHeadline>
          <RadioGroup
            id="mascot-radio-group"
            aria-labelledby="choose-a-mascot"
            name="radio-group2"
            selectedValue={group2}
            onChange={setGroup2}
            options={options2}
          />
        </>
      );
    }}
  </Story>
</Canvas>
