## Usage

```tsx
import { TextDisplay2, TextBody } from '@cds/web/components'
import { ThemeProvider } from '@cds/theme'

// Text styles are provided through the theme provider
<ThemeProvider scale="xSmall">
  <TextDisplay2 as="h1" color="primary" align="center">Coinbase Design System</TextDisplay2>
  <TextTitle2 as="p" wrap >Buy and sell popular <a href="/currencies">digital currencies</a></TextTitle2>
  <TextBody as="code" display="inline" tnum>2020</TextBody>
  <ul>
    <li><TextBody as="p">BTC</TextBody></li>
    <li><TextBody as="p">ETH</TextBody></li>
    <li><TextBody as="p">XRP</TextBody></li>
    <li><TextBody as="p">DAI</TextBody></li>
  </ul>
<ThemeProvider>
```
