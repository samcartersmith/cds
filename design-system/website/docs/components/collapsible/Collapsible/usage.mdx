### Basic Collapsible
```jsx live
function BasicCollapsible() {
  const [collapsed, { toggle }] = useToggler(true);

  return (
    <ExampleWithThemeToggles>
      <Button onPress={toggle}>Click me!</Button>
      <Collapsible collapsed={collapsed}>
        <TextInput
          compact
          type="number"
          step="0.01"
          label="Amount"
          placeholder="8293323.23"
          suffix="USD"
        />
      </Collapsible>
    </ExampleWithThemeToggles>
  )
}
```
