# PopoverMenu Prototype

## Default

```jsx live
function Default() {
  const [value, setValue] = useState('');
  const [visible, togglePopoverMenuVisibility] = useToggler(false);

  const priceOptions = [
    '666',
    '2387',
    '4542',
    '54534',
    '5435',
    '77',
    '123',
    '4345345',
    '6544',
    '874',
    '675765',
    '65655',
  ];

  const handleMenuValueChange = useCallback(
    (newValue) => {
      setValue(newValue);
    },
    [setValue],
  );

  return (
    <VStack minHeight={500} spacing={4} background="background">
      <PopoverMenu
        width={200}
        value={value}
        onChange={handleMenuValueChange}
        openMenu={togglePopoverMenuVisibility.toggleOn}
        closeMenu={togglePopoverMenuVisibility.toggleOff}
        visible={visible}
      >
        <PopoverTrigger>
          <IconButton transparent name="more" variant="secondary" />
        </PopoverTrigger>
        {priceOptions.map((option) => (
          <SelectOption
            value={option}
            key={option}
            title={option}
            description="BTC"
            testID={`option-${option}`}
            selected={value === option}
          />
        ))}
      </PopoverMenu>
    </VStack>
  );
}
```
